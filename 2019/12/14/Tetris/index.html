上面的行下移
private void RowDown(int currentRow)
{
    for (int row = currentRow; row < height; row++)
    {
        for (int col = 0; col < width; col++)
        {
            if (grid[col, row] != null)
            {
                grid[col, row - 1] = grid[col, row];
                grid[col, row] = null;
                grid[col, row - 1].transform.position -= new Vector3(0, 1, 0);
            }
        }
    }
}
